function unitMovementCheck(averageCodingScore, averageHukumuScore, cutoff) { return new Promise((resolve, reject) => { setTimeout(() => { const scores = [averageCodingScore, averageHukumuScore]; const total = scores.reduce((acc, score) => acc + score, 0); const average = total / scores.length; if (average >= cutoff) { resolve(average); } else { reject("Sorry you haven’t cleared the final cutoff."); } Document }, 2000); }); } // Example usage: unitMovementCheck(85, 78, 80) .then((average) => console.log(`Final average score: ${average}`)) .catch((error) => console.log(error));
